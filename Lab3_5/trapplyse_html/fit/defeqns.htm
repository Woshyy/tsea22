********** Mapped Logic **********
FDCPE_clr: FDCPE port map (clr,u.PIN,clk,reset,'0');
FDCPE_rco: FDCPE port map (rco,rco_D,clk,'0',reset);
     rco_D <= ((clr.LFBK AND NOT rco.LFBK AND u.PIN)
      OR (NOT counter(0).LFBK AND clr.LFBK AND NOT counter(3).LFBK AND
      u.PIN AND NOT tidkvar(1).PIN AND NOT tidkvar(2).PIN));
FDCPE_tidkvar0: FDCPE port map (tidkvar(0),tidkvar_D(0),clk,reset,'0');
     tidkvar_D(0) <= ((NOT u.PIN)
      OR (counter(0).LFBK AND clr.LFBK)
      OR (clr.LFBK AND NOT counter(3).LFBK AND NOT tidkvar(1).PIN AND
      NOT tidkvar(2).PIN));
FDCPE_tidkvar1: FDCPE port map (tidkvar(1),tidkvar_D(1),clk,reset,'0');
     tidkvar_D(1) <= ((NOT clr AND u.PIN)
      OR (counter(1).LFBK AND u.PIN AND tidkvar(0).PIN)
      OR (NOT counter(1).LFBK AND u.PIN AND NOT tidkvar(0).PIN AND
      tidkvar(2).PIN)
      OR (NOT counter(1).LFBK AND u.PIN AND NOT tidkvar(0).PIN AND
      tidkvar(3).PIN));
FDCPE_tidkvar2: FDCPE port map (tidkvar(2),tidkvar_D(2),clk,reset,'0');
     tidkvar_D(2) <= ((NOT clr AND u.PIN)
      OR (counter(2).LFBK AND u.PIN AND tidkvar(0).PIN)
      OR (counter(2).LFBK AND u.PIN AND tidkvar(1).PIN)
      OR (NOT counter(2).LFBK AND u.PIN AND NOT tidkvar(0).PIN AND
      NOT tidkvar(1).PIN AND tidkvar(3).PIN));
FDCPE_tidkvar3: FDCPE port map (tidkvar(3),tidkvar_D(3),clk,reset,'0');
     tidkvar_D(3) <= ((NOT u.PIN)
      OR (clr.LFBK AND NOT counter(3).LFBK)
      OR (NOT counter(0).LFBK AND clr.LFBK AND NOT tidkvar(1).PIN AND
      NOT tidkvar(2).PIN));
FTCPE_u: FTCPE port map (u,u_T,clk,reset,'0');
     u_T <= ((NOT rco AND ce.LFBK)
      OR (x0 AND NOT x0_last)
      OR (NOT x0 AND x0_last)
      OR (x1 AND NOT x1_last)
      OR (NOT x1 AND x1_last));
FDCPE_x0_last: FDCPE port map (x0_last,x0,clk,reset,'0');
FDCPE_x1_last: FDCPE port map (x1_last,x1,clk,reset,'0');
